50  {

52  exponent++;
53  }
54  }

90  }

94  {
95  inv_ibuf[ibufsiz] = (CHAR)((ipart % 10) + 48);
96  ipart /= 10;
97 
98  ibufsiz++;
99  }
100 
101

103  {
104  fpart *= 10;
105

107 
108  inv_fbuf[fbufsiz] = (CHAR)(frac_char + 48);
109 
110  fbufsiz++;
111  fpart -= frac_char;
112  }

117  exponent /= 10;

156  }
157  }

161  }
